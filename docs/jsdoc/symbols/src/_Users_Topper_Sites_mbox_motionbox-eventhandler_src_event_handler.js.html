151             then this function should should take an optional opts argument

156             @param {HTMLElement} elem The html element to subscribe to
157             @param {String} type The type of event (eg 'click')

161             @returns {Object} An event subscription
162             @function

349     /**
350         executes an array of functions sending the event to the function

352     var callFunctions = function (functionsToCall, evt) {
353         while (functionsToCall.length > 0) {

358     var wrap = function(func, wrapper) {
359         var __method = func;
360         __method.bind = function() {
361             if (arguments.length < 2 && Object.isUndefined(arguments[0])) return this;
362             var __method = this, args = Array.prototype.slice.call(arguments), object = args.shift();
363             return function() {
364               return __method.apply(object, args.concat(Array.prototype.slice.call(arguments)));
365             }
366         };
367         
368         return function() {
369           return wrapper.apply(this, [__method.bind(this)].concat(Array.prototype.slice.call(arguments)));
370         }
371     };

373     /**
374         defers function for later execution
375     */
376     var deferFunctions = function (functionsToCall, evt) {
377         var func;
378         while (functionsToCall.length > 0) {
379             func = wrap(functionsToCall.pop(), function (orig) {
380                 orig(evt);
381             });
382             setTimeout(func, 0);
383         }
384     };

386     /** if there is a listener defined for the evtType, then
387         loop through those rules and compare them to target
388         bad CSS selectors can throw up really bad JS errors,
389     */
390     var callFunctionsFromRules = function (target, evtType, evt) {
391         if (!subscriptions.rules[evtType]) {
392             return;
393         }
394         var subscriptionsToCall = [];
395         for (prop in subscriptions.rules[evtType]) {
396             if (subscriptions.rules[evtType].hasOwnProperty(prop) && target.match(prop)) {
397                 subscriptionsToCall = subscriptionsToCall.concat(subscriptions.rules[evtType][prop]);
398             }
399         }
400         eachElement(subscriptionsToCall, function (sub) {
401             sub.callFunctions(evt);
402         });
403     };

405     /** go to the subscriptions.ids object and grab an array of all the functions that are subscribed to
406         the eventType evtType... so subscriptions.ids[targetId][evtType] which will be an array of functions
407     */
408     var callFunctionsFromIdOrObject = function (specifierType, targetId, evtType, evt) {
409         var returnArray = [];
410         var deferArray = [];
411         var subscriptionTarget = subscriptions[specifierType][targetId];
412         if (subscriptionTarget && subscriptionTarget[evtType]) {
413             returnArray = returnArray.concat(subscriptionTarget[evtType]);
414         }
415         eachElement(returnArray, function (sub) {
416             sub.callFunctions(evt);
417         });
418     };
419     
420     /** same as functionsFromId, but uses all classes on the target object and looks in
421         subscriptions.classes object
422     */
423     var callFunctionsFromClasses = function (targetClasses, evtType, evt) {
424         var subscriptionsToCall = [];
425         var functionsToDefer = [];
426         var classObject;
427         for (var index = 0, classLen = targetClasses.length; index < classLen; ++index) {
428             classObject = subscriptions.classes[targetClasses[index]];
429             if (classObject && classObject[evtType]) {
430                 subscriptionsToCall = subscriptionsToCall.concat(classObject[evtType]);
431             }
432         }
433         eachElement(subscriptionsToCall, function (sub) {
434             sub.callFunctions(evt);
435         });
436     };

438     /** given an element and an event type, call the functions held in the 
439         subscriptions object
440     */
441     var functionsFromElementAndEvent = function (targetElement, evt, opts) {
442         if (!targetElement) {
443             return;
444         }
445         
446         opts = opts || {};
447         var evtType;
448         if (opts.eventType) {
449             evtType = opts.eventType;
450         } else {
451             evtType = evt.type;
452         }
453         
454         if(targetElement.__MotionboxEventHandlerMaker) {
455             callFunctionsFromIdOrObject("objects", targetElement.__MotionboxEventHandlerMaker, evtType, evt);
456             if (!Object.isElement(targetElement)) {
457                 return;
458             }
459         }
460         
461         if (targetElement.id) {
462             callFunctionsFromIdOrObject("ids", targetElement.id, evtType, evt);
463         }
464         
465         if (targetElement.className) {
466             var targetClasses = targetElement.className.split(" ");
467             callFunctionsFromClasses(targetClasses, evtType, evt);
468         }
469         callFunctionsFromRules(targetElement, evtType, evt);
470     
471         //recursively call self walking up the tree
472         if (targetElement != window && targetElement != document && targetElement.parentNode) {
473             var upTreeNode = targetElement.parentNode;
474             if (upTreeNode && upTreeNode.tagName && upTreeNode.tagName != "HTML") {
475                 functionsFromElementAndEvent(upTreeNode, evt, opts);
476             }
477         }
478     };

509     /** 
510      The Object that holds a subcription.
511      Contains a specifier (object, id, class), type (eg 'click'), and functions
512      @constructor
513     */
514     var Subscription = function (specifier, eventType, funcs, opts) {
515         funcs = makeArray(funcs);
516         
517         this.specifier = specifier;
518         this.eventType = eventType;
519         this.funcs = funcs;
520         this.opts = opts || {};
521         this.calculateSpecifierType();
522         this.cache();
523         return this;
524     };
525     
526     Subscription.prototype = {
527         callFunctions: function (evt) {
528             if (this.opts.defer) {
529                 deferFunctions(this.funcs, evt);
530             } else {
531                 callFunctions(this.funcs, evt);
532             }
533         },
534         calculateSpecifierType: function () {
535             if (isObject(this.specifier)) {
536                 this.specifierType = "objects";
537                 this.specifier = getSubscriptionMarker(this.specifier);
538             } else {
539                 if (typeof this.specifier != "string") {
540                     throw new Error("specifier was neither an object nor a string");
541                 }
542                 if (isId(this.specifier)) {
543                     this.specifier = this.specifier.sub(/#/, "");
544                     this.specifierType = "ids";
545                 }
546                 if (isClass(this.specifier)) {
547                     this.specifierType = "classes";
548                     this.specifier = this.specifier.sub(/\./, "");
549                 }
550                 if (!this.specifierType) {
551                     this.specifierType = "rules";
552                 }
553             }
554         },
555         _cacheForRules: function () {
556             if (!subscriptions.rules[this.eventType]) {
557                 subscriptions.rules[this.eventType] = {};
558             }
559             var specifierObject = subscriptions.rules[this.eventType];
560             
561             if (!specifierObject[specifier]) {
562                 specifierObject[specifier] = [];
563             }
564             specifierObject[specifier].push(this);
565         },
566         cache: function () {
567             if (this.specifierType == "rules") {
568                 this._cacheForRules();
569             } else {
570                 if (!subscriptions[this.specifierType][this.specifier]) {
571                     subscriptions[this.specifierType][this.specifier] = {};
572                 }
573                 var specifierObject = subscriptions[this.specifierType][this.specifier];
574                 if (!specifierObject[this.eventType]) {
575                      specifierObject[this.eventType] = [];
576                  }
577                 
578                 specifierObject[this.eventType].push(this);
579             }
580         },
581         _uncacheForRules: function () {
582             var specifierObject = subscriptions.rules[this.eventType];
583             specifierObject.splice(specifierObject.indexOf(this), 1);
584         },
585         uncache: function () {
586             if (this.specifierType == "rules") {
587                 this._uncacheForRules();
588             } else {
589                 var specifierObject = subscriptions[this.specifierType][this.specifier][this.eventType];
590                 specifierObject.splice(specifierObject.indexOf(this), 1);
591             }
592         }
593     };

595     var SubscriptionSet = function (specifiers, evtTypes, funcs, opts) {
596         specifiers = makeArray(specifiers);
597         evtTypes = makeArray(evtTypes);
598         
599         this.specifiers = specifiers;
600         this.evtTypes = evtTypes;
601         this.funcs = funcs;
602         this.opts = opts || {};
603         this.subscriptions = [];
604         this.createSubscriptions();
605         return this;
606     };
607     
608     SubscriptionSet.prototype = {
609         createSubscriptions: function () {
610             for (var i = 0; i < this.specifiers.length; ++i) {
611                 for (var j = 0; j < this.evtTypes.length; ++j) {
612                     this.subscriptions.push(new Subscription(this.specifiers[i], this.evtTypes[j], this.funcs, this.opts));
613                 }
614             }
615         },
616         unsubscribe: function () {
617             eachElement(this.subscriptions, function (subscription) {
618                 subscription.uncache();
619             });
620         }
621     };

623     /** institue the subscriber:  '#' indicates an id, "." indicates a class, any other string is
624         considered a CSS Selector.  You may also pass in an object (or DomElement).
625         subscribe with:
626         @example
627           MBX.EventHandler.subscribe(".myClass", "click", function (){ alert('hi'); });
628         @example
629           MBX.EventHandler.subscribe("p#blah.cool", "click", function(evt) {console.dir(evt);});
630         @example
631           var someObj = {};
632           // the below will use a setTimeout to fire the function when "myEvent" is fired on someObj
633           MBX.EventHandler.subscribe(someObj, "myEvent", function () {alert('hi'); }, {defer: true});

635         events may be custom events
636         
637         @param {String or Object or Array} specifiers the Object, class, id or CSS selector that you want to subscribe to (or array of any of those)
638         @param {String or Array} evtTypes the types of events you want to subscribe to (these can be arbitrary to allow custom events)
639         @param {Function or Array} funcs the functions you want to be called with this subscription
640         @param {Object} opts Right now only takes a "defer" option which will fire functions with setTimeout
641

647         @name MBX.EventHandler.subscribe

673         @param {String} evt the Event to fire

675         
676         @example

702     /** Accepts functions that will be fired as soon as the dom is ready (using prototypes dom:loaded event)
703         By default, we fire onDomReady events using setTimeout
704         If the dom:loaded event has already ocurred, we simply call the function
705         @param {Function or Array} funcs the function(s) to be fired at the Dom Ready event
706         @param {Object} opts (optional) add { defer: false } to *not* fire the function using a setTimeout
707         @returns a handler object that can be used to unsubscribe
708         
709         @name MBX.EventHandler.onDomReady
710         @function
711     */
712     self.onDomReady = function (funcs, opts) {
713         opts = opts || {};
714         if (typeof opts.defer == 'undefined') {
715             opts.defer = true;
716         }
717         
718         funcs = makeArray(funcs);
719         if (domReadyAlreadyFired) {
720             if (opts.defer) {
721                 deferFunctions(funcs, "dom:loaded");
722             } else {
723                 callFunctions(funcs, "dom:loaded");
724             }
725         } else {
726             return MBX.EventHandler.subscribe(document, "dom:loaded", funcs, { defer: opts.defer });
727         }
728     },
