 63 /**  @class EventHandler
 64      @name MBX.EventHandler */

 66     /** 
 67         all the standard events we want to listen to on document.
 68         please note that 'change' and 'blur' DO NOT BUBBLE in IE - so you will need to do something
 69         extra for the Microsoft browsers
 70     */
 71     var stdEvents = ["click", "mouseout", "mouseover", "keypress", "change", "blur", "focus"];
 72     
 73     /** an object with all the event listeners we have listed by eventType
 74         gets filled in on init
 75     */
 76     var eventListeners = {};
 77     
 78      /** Event bubbles up to the document where the listeners are and fires this function
 79          if the target element matches anything in the subscribers object then the functions fire
 80          it continues to go all the way up the tree
 81          @function
 82      */
 83      var handleEvent = function (evt) {
 84          //for debug uncomment out the below
 85          //console.dir(evt);
 86          //console.log(Event.element(evt));
 87          //evt.isConsumed = false;
 88          var targetElement;
 89          //the below fixes an intermittent prototype JS error
 90          if (Event && evt) {
 91              try {
 92                  targetElement = Event.element(evt);
 93              } catch (e) {
 94              }
 95          }
 96          if (targetElement) {
 97              functionsFromElementAndEvent(targetElement, evt);
 98          }
 99     };
100     
101     /** subscribe to the listeners
102     */
103     stdEvents.each(function (evtType) {
104         eventListeners[evtType] = document.observe(evtType, handleEvent);
105     });
106     
107     document.observe("dom:loaded", function () {
108         MBX.EventHandler.fireCustom(document, "dom:loaded");
109     });

111     /** this holds the actual subscriptions in the form
112     
113         @example
114           ids: {
115               myId: {
116                       myEventType: [function, function, function]
117                  }
118           }
119           
120         same for classes and objects (objects add a unique identifier);
121         rules however is the opposite (for speed sake)
122         so:
123         
124         @example
125           rules: {
126               eventType: {
127                   "my rule": [function, function, function]
128               }
129           }
130     */
131     var subscriptions = {
132         ids: {},
133         classes: {},
134         rules: {},
135         objects: {}
136     };
137     
138     var subscriptionMarker = 1;
139     
140     /**
141         executes an array of functions sending the event to the function
142     */
143     var callFunctions = function (functionsToCall, evt) {
144         while (functionsToCall.length > 0) {
145             functionsToCall.pop()(evt);
146         }
147     };
148     
149     /**
150         defers function for later execution
151     */
152     var deferFunctions = function (functionsToCall, evt) {
153         var func;
154         while (functionsToCall.length > 0) {
155             func = functionsToCall.pop();
156             var thisFire = function() {
157                 func(evt);
158             }
159             setTimeout(thisFire, 0);
160         }
161     };
162     
163     /** if there is a listener defined for the evtType, then
164         loop through those rules and compare them to target
165         bad CSS selectors can throw up really bad JS errors,
166     */
167     var callFunctionsFromRules = function (target, evtType, evt) {
168         if (!subscriptions.rules[evtType]) {
169             return;
170         }
171         var functionsToCall = [];
172         var functionsToDefer = [];
173         for (prop in subscriptions.rules[evtType]) {
174             if (subscriptions.rules[evtType].hasOwnProperty(prop) && target.match(prop)) {
175                 functionsToCall = functionsToCall.concat(subscriptions.rules[evtType][prop]);
176                 if (subscriptions.rules[evtType][prop]['deferrable']) {
177                     functionsToDefer = functionsToDefer.concat(subscriptions.rules[evtType][prop]['deferrable']);
178                 }
179             }
180         }
181         callFunctions(functionsToCall, evt);
182         deferFunctions(functionsToDefer, evt);
183     };
184     
185     /** go to the subscriptions.ids object and grab an array of all the functions that are subscribed to
186         the eventType evtType... so subscriptions.ids[targetId][evtType] which will be an array of functions
187     */
188     var callFunctionsFromIdOrObject = function (specifierType, targetId, evtType, evt) {
189         var returnArray = [];
190         var deferArray = [];
191         var subscriptionTarget = subscriptions[specifierType][targetId];
192         if (subscriptionTarget && subscriptionTarget[evtType]) {
193             returnArray = returnArray.concat(subscriptionTarget[evtType]);
194             if (subscriptionTarget[evtType]['deferrable']) {
195                 deferArray = deferArray.concat(subscriptionTarget[evtType]['deferrable']);
196             }
197         }
198         callFunctions(returnArray, evt);
199         deferFunctions(deferArray, evt);
200     };
201     
202     /** same as functionsFromId, but uses all classes on the target object and looks in
203         subscriptions.classes object
204     */
205     var callFunctionsFromClasses = function (targetClasses, evtType, evt) {
206         var functionsToCall = [];
207         var functionsToDefer = [];
208         var classObject;
209         targetClasses = $A(targetClasses);
210         for (var index = 0, classLen = targetClasses.length; index < classLen; ++index) {
211             classObject = subscriptions.classes[targetClasses[index]]
212             if (classObject && classObject[evtType]) {
213                 functionsToCall = functionsToCall.concat(classObject[evtType]);
214                 if (classObject[evtType]['deferrable']) {
215                     functionsToDefer.concat(classObject[evtType]['deferrable']);
216                 }
217             }
218         }
219         callFunctions(functionsToCall, evt);
220         deferFunctions(functionsToDefer, evt);
221     };
222     
223     /** given an element and an event type, call the functions held in the 
224         subscriptions object
225     */
226     var functionsFromElementAndEvent = function (targetElement, evt) {
227         var evtType = evt.type;
228         if (!targetElement) {
229             return;
230         }
231         
232         if(targetElement.__MotionboxEventHandlerMaker) {
233             callFunctionsFromIdOrObject("objects", targetElement.__MotionboxEventHandlerMaker, evtType, evt);
234             if (!Object.isElement(targetElement)) {
235                 return;
236             }
237         }
238         
239         if (targetElement.id) {
240             callFunctionsFromIdOrObject("ids", targetElement.id, evtType, evt);
241         }
242         
243         if (targetElement.className) {
244             var targetClasses = Element.classNames(targetElement);
245             callFunctionsFromClasses(targetClasses, evtType, evt);
246         }
247         callFunctionsFromRules(targetElement, evtType, evt);
248     
249         //recursively call self walking up the tree
250         if (targetElement != window && targetElement != document && targetElement.parentNode) {
251             var upTreeNode = targetElement.parentNode;
252             if (upTreeNode && upTreeNode.tagName && upTreeNode.tagName != "HTML") {
253                 functionsFromElementAndEvent($(upTreeNode), evt);
254             }
255         }
256     };
257     
258     /** handle the creation of ID or class based subscriptions for a single
259         specifier arrays of types and functions
260     */
261     var createIdClassorObjectSubscription = function(specifierType, specifier, evtTypes, funcs, opts) {
262         var subscriptionArray = [];
263         if (!subscriptions[specifierType][specifier]) {
264             subscriptions[specifierType][specifier] = {};
265         }
266         var specifierObject = subscriptions[specifierType][specifier];
267         evtTypes.each(function (evtType) {
268              if (!specifierObject[evtType]) {
269                  specifierObject[evtType] = [];
270              }
271              if (opts.defer && !specifierObject[evtType]['deferrable']) {
272                  specifierObject[evtType]['deferrable'] = [];
273              }
274             
275             funcs.each(function (func) {
276                 if (opts.defer) {
277                     specifierObject[evtType]['deferrable'].push(func);
278                 } else {
279                     specifierObject[evtType].push(func);
280                 }
281                 subscriptionArray.push({'specifierType': specifierType, 'eventType': evtType, 'func': func, 'specifier': specifier});
282             });
283         });
284         return subscriptionArray;
285     };
286     
287     /** handle a CSS selector based subscription for a single specifier and arrays of types and functions
288     */
289     var createRulesSubscription = function(specifier, evtTypes, funcs, opts) {
290         var subscriptionArray = [];
291         evtTypes.each(function (evtType) {
292             if (!subscriptions.rules[evtType]) {
293                 subscriptions.rules[evtType] = {};
294             }
295             var specifierObject = subscriptions.rules[evtType];
296             
297             if (!specifierObject[specifier]) {
298                 specifierObject[specifier] = [];
299             }
300             if (opts.defer && !specifierObject[specifier]['deferrable']) {
301                 specifierObject[specifier]['deferrable'] = [];
302             }
303             funcs.each(function (func) {
304                 if (opts.defer) {
305                     specifierObject[specifier]['deferrable'].push(func);
306                 } else {
307                     specifierObject[specifier].push(func);
308                 }
309                 subscriptionArray.push({'specifierType': 'rules', 'eventType': evtType, 'func': func, 'specifier': specifier});
310             }); //each function
311         }); // each event type
312         return subscriptionArray;
313     };

315     // utility functions
316     var isArray = function (obj) {
317         if (obj) {
318             return obj.constructor == Array;
319         }
320     };

322     var isId = function (specifierString) {
323         return /^\#[\w-]+$/.test(specifierString);
324     };
325     
326     var isClass = function (specifierString) {
327         return /^\.[\w-]+$/.test(specifierString);
328     };
329     
330     var isObject = function (specifier) {
331         return typeof specifier == "object";
332     };
333     
334     var getSubscriptionMarker = function (obj) {
335         if (!obj.__MotionboxEventHandlerMaker) {
336             obj.__MotionboxEventHandlerMaker = subscriptionMarker++;
337         }
338         return obj.__MotionboxEventHandlerMaker;
339     };

341     var browserLikeEventExtender = {
342         preventDefault: function () {},
343         stopPropagation: function () {},
344         pageX: 0,
345         pageY: 0,
346         clientX: 0,
347         clientY: 0
348     };

350     var CustomEvent = function (theTarget, evt, opts) {
351         this.type = evt;
352         this.target = theTarget;
353         this.srcElement = theTarget;
354         this.eventName = evt;
355         this.memo = {};
356         Object.extend(this, opts);
357         for (prop in browserLikeEventExtender) {
358             if (browserLikeEventExtender.hasOwnProperty(prop)) {
359                 if (!this[prop]) {
360                     this[prop] = browserLikeEventExtender[prop];
361                 }
362             }
363         }
364         if (Prototype.Browser.IE) {
365             Event.extend(this);
366         }
367     };
368         
369     if (!Prototype.Browser.IE) {
370         (function () {
371             var methods = Object.keys(Event.Methods).inject({ }, function(m, name) {
372                 m[name] = Event.Methods[name].methodize();
373                 return m;
374               });
375     
376             CustomEvent.prototype = CustomEvent.prototype || document.createEvent("HTMLEvents").__proto__;
377             Object.extend(CustomEvent.prototype, methods);
378         })();
379     }

381     if (Prototype.Browser.IE) {
382         var destroyObservers = function () {
383             stdEvents.each(function (evtType) {
384                 document.stopObserving(evtType, handleEvent);
385             });
386         };
387         
388         window.attachEvent('onunload', destroyObservers);
389     }

391     return /** @lends MBX.EventHandler */ {

394         /** institue the subscriber:  '#' indicates an id, "." indicates a class, anything else is considered
395             a CSS Selector
396             subscribe with:
397             @example
398               MBX.EventHandler.subscribe(".myClass", "click", function (){ alert('hi'); });
399             or:
400             @example
401               MBX.EventHandler.subscribe("p#blah.cool", "click", function(evt) {console.dir(evt);});
402             events may be custom events
403             
404             @param {String} specifiers the class, id or CSS selector that you want to subscribe to
405             @param {String or Array} evtTypes the types of events you want to subscribe to
406             @param {Function or Array} funcs the functions you want to be called with this subscription
407             @param {Object} opts Right now only takes a "defer" option which will fire functions with setTimeout
408             
409             @returns A handler object that can be used to unsubscribe
410             
411             @see MBX.EventHandler.fireCustom.
412             returns an object you can use to unsubscribe
413         */
414         subscribe: function (specifiers, evtTypes, funcs, opts) {
415             if (!isArray(specifiers)) {
416                 specifiers = [specifiers];
417             }
418             if (!isArray(evtTypes)) {
419                 evtTypes = [evtTypes];
420             }
421             if (!isArray(funcs)) {
422                 funcs = [funcs];
423             }
424             opts = opts || {};
425             var referenceArray = [];
426     
427             specifiers.each(function (specifier) {
428                 var specifierType;
429                 if (isObject(specifier)) {
430                     specifierType = "objects";
431                     specifier = getSubscriptionMarker(specifier);
432                 } else {
433                     if (typeof specifier != "string") {
434                         throw new Error("specifier was neither an object nor a string");
435                     }
436                     if (isId(specifier)) {
437                         specifier = specifier.sub(/#/, "");
438                         specifierType = "ids";
439                     }
440                     if (isClass(specifier)) {
441                         specifierType = "classes";
442                         specifier = specifier.sub(/\./, "");
443                     }
444                 }
445                 
446                 //check if it matched id or class
447                 if (specifierType) {
448                     referenceArray = referenceArray.concat(createIdClassorObjectSubscription(specifierType, specifier, evtTypes, funcs, opts));
449                 } else {
450                     // we assume that anything not matching a class, id or object is a css selector rule
451                     referenceArray = referenceArray.concat(createRulesSubscription(specifier, evtTypes, funcs, opts));
452                 } //end to rules handling
453             }); // each specifier
454             // return the array that can be used to unsubscribe
455             return referenceArray;
456         },
457         
458         /** Unsubscribe a previous subscribed handler
459             @param {Object or Array} handlerObjects the handler objects that were originally passed to the
460                                                     subscriptions
461             @example
462               var handlerObj = MBX.EventHandler.subscribe("#blah", "click", function () {alert('hi')!});
463               MBX.EventHandler.unsubscribe(handlerObj) // the subscription above is now removed
464         */
465         unsubscribe: function (handlerObjects) {
466             handlerObjects.each(function (handlerObject) {
467                 if (!(handlerObject.specifierType && handlerObject.eventType && handlerObject.specifier) || typeof handlerObject.func != 'function') {
468                     throw new Error('bad unsubscribe object passed to EventHandler.unsubscribe');
469                 }
470                 if (handlerObject.specifierType != "rules") {
471                     var locator = subscriptions[handlerObject.specifierType][handlerObject.specifier][handlerObject.eventType];
472                     for (var i = 0, funcLen = locator.length; i < funcLen; ++i) {
473                         if (locator[i] == handlerObject.func) {
474                             subscriptions[handlerObject.specifierType][handlerObject.specifier][handlerObject.eventType].splice(i, 1);
475                         }
476                     }
477                 } else {
478                     var locator = subscriptions[handlerObject.specifierType][handlerObject.eventType][handlerObject.specifier];
479                     for (var i = 0, funcLen = locator.length; i < funcLen; ++i) {
480                         if (locator[i] == handlerObject.func) {
481                             subscriptions[handlerObject.specifierType][handlerObject.eventType][handlerObject.specifier].splice(i, 1);
482                         }
483                     }
484                 }
485     
486             });
487     
488             return true;
489         },
490     
491         /** fire a custom event of your choosing. Will notify any subscribers to that evt
492             You can also attach a payload to the event that will be added to the events
493             @param {DomElement} theTarget A dom element to fire the event on
494             @param {String} evt the Event to fire
495             @param {Object} opts (optional) the attributes to be attached to the event
496             
497             @example
498               MBX.EventHandler.fireCustom($('element'), 'mycustomeevent');
499               
500             @example
501               MBX.EventHandler.fireCustom($("element"), 'mycustomevent', {
502                   theseAttributes: "will be attached to the event"
503               });
504         */
505         fireCustom: function (theTarget, evt, opts) {
506             opts = opts || {};
507             if (Object.isElement(theTarget)) {
508                 var theEvent = new CustomEvent(theTarget, evt, opts);
509                 Event.extend(theEvent);
510                 handleEvent(theEvent);
511             } else {
512                 callFunctionsFromIdOrObject("objects", getSubscriptionMarker(theTarget), evt, opts);
513             }
514         },
515         
516         /** these functions will be fired as soon as the dom is ready (using prototypes dom:ready event)
517             @param {Function or Array} funcs the function(s) to be fired at the Dom Ready event
518         */
519         onDomReady: function (funcs) {
520             if (!Object.isArray(funcs)) {
521                 funcs = [funcs];
522             }
523             return MBX.EventHandler.subscribe(document, "dom:loaded", funcs);
524         },

532         },
533         debugSubscriptions: function () {
